# Alternative Email Configurations

## Option 1: Outlook/Hotmail
spring.mail.host=smtp-mail.outlook.com
spring.mail.port=587
spring.mail.username=your-outlook-email@outlook.com
spring.mail.password=your-outlook-password
spring.mail.properties.mail.smtp.auth=true
spring.mail.properties.mail.smtp.starttls.enable=true

## Option 2: Yahoo Mail
spring.mail.host=smtp.mail.yahoo.com
spring.mail.port=587
spring.mail.username=your-yahoo-email@yahoo.com
spring.mail.password=your-yahoo-app-password
spring.mail.properties.mail.smtp.auth=true
spring.mail.properties.mail.smtp.starttls.enable=true

## Option 3: Mailtrap (Testing Only)
spring.mail.host=smtp.mailtrap.io
spring.mail.port=2525
spring.mail.username=your-mailtrap-username
spring.mail.password=your-mailtrap-password
